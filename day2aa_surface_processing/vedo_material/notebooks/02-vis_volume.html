Volume data in vedo#

from vedo import Volume, settings, dataurl
from vedo.applications import RayCastPlotter

settings.default_backend = "vtk"  # or k3d, ipyvtk, or 2d

vol = Volume(dataurl + "embryo.tif")
vol.mode(1).cmap("jet")  # change appearance
