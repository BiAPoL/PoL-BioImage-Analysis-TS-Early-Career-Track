from skimage import data
from vedo import Volume, show

settings.default_backend = "vtk"

# Load some 3D multichannel array with dimensions provided in (z, c, y, x) order. 
# Each voxel has a size of (0.29 0.26 0.26) micrometer. 
# Channel 0 contains cell membranes, channel 1 contains nuclei.
cells = data.cells3d()[:,1,:,:]

vol = Volume(cells).permute_axes(2,1,0).spacing([0.26, 0.26, 0.29])
vol.smooth_gaussian([3,3,1])
